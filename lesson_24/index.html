Задача №1

Що потрапить в консоль?

let someVar = 0;
++someVar;

if (someVar) {
console.log(someVar);
}

Розв'язок:

Якщо someVar = 0, і в наступному рядку до нього застосовується префіксний інкремент "++", який збільшує число на 1, то матимемо:
someVar = 0 + 1 = 1

Відповідь: 1

За допомогою циклу FOR виведіть в консоль 10 рядків:

Пункт №1
Пункт №2
і т.д.

Розв'язок:

Створюємо змінну listItem зі значенням: "Пункт №", а номер пункту буде задаватись через цикл "for".
Таким чином цикл "for" буде використовувати змінну "і", значення якої буде починатись з 1, закінчується цикл при умові "i <= 10", з кроком "++і"="+1" .

let listItem = "Пункт №"

for (let i = 1; i <= 10; ++i) {
console.log(listItem + i)
}

Задача №3

Що потрапить в консоль ?

if (2 * 20 <= 10 || 30 / 2 < 5 && 10 <="10" || 20==="20" ) {
console.log('000');
}

Розв'язок:

І варіант. Для того щоб спростити розв'язок можемо виокремити у душки дії у "рівнянні"
Тобто, тепер почергово розв'язуємо:
(((2 * 20) <= 10) || (((30 / 2) < 5) && (10 <="10" )) || (20==="20" ))
1. ((2 * 20) <= 10)=40 <=10 - false
2. ((30 / 2) < 5)=15 < 5 - false
3. (10 <="10" ) - true
4. (20==="20") - false (адже "20" це текст, і оператор "===" не перетворює в число)
5. Перевіряємо оператор &&, тобто: дію 2. && дію 3.
Маємо: false && true - false (адже не обидва значення true)
6. Перевіряємо оператор ||, тобто: дію 1. || дію 5. || дію 4.
Маємо: false || false || false - false (адже не має значення true, тому отримаємо останнє false)

ІІ варіант. Розв'язуємо по пріоритетності операторів, у цьому "рівнянні" маємо такий порядок:
1. *, /;
2. <=, <;
3. ===;
4. &&;
5. ||.
1. 2 * 20 = 40
30 / 2 = 15
Маємо: (40 <= 10 || 15 < 5 && 10 <="10" || 20==="20" )
2. 40 <= 10 - false
15 < 5 - false
10 <="10" - true
Маємо: (false || false && true || 20==="20" )
3. 20 === "20" - false (адже "20" це текст, і оператор "===" не перетворює в число)
Маємо: (false || false && true || false )
4. false && true false (адже не обидва значення true)
Маємо: (false || false || false )
5. false || false || false - false (адже не має значення true, тому отримаємо останнє false)

console.log(helloTrue)

else {

Задача №4

Створіть функцію, яка повертає результат ділення числа a на число b з додаванням рядка "Результат ділення: " Викличте функцію передаючи різні значення, у тому числі не передаючи зовсім. Функція не має повертати NaN, Infinite або помилку

Розв'язок:

Задаємо для початку функцію "showMessage" для виводу в консоль текстового повідомлення зі зміною "text" з значенням за замовченням "Нічого не вказано"

Задаємо функцію "checkTypeNumber" для перевірки чи вказаний тип параметрів "someVar" для функції є числовим:
(typeof someVar == "number"), якщо так, то тоді повертаємо значення "someVar" для подальшого ділення, якщо ні - false.

Далі задаємо функцію безпосередньо для ділення з назвою "calcDivision" з параметрами "a" = 0 (за замовченням), "b" = 0 (за замовченням)
В функції задаємо зміну "divisionText" = з значенням "Результат ділення: " (для виводу тексту перед результатом ділення)
Та зміну "divisionTextError" = з значенням "ділення неможливе" (для виводу тексту, якщо ділимо на 0, або не вказані число/-а, тобто результат Infinite, NaN і т.д.)

Перевіряємо чи обидва наші типи параметрів одночасно є числовими за допомогою раніше створеної функції "checkTypeNumber":
if (checkTypeNumber(a) && checkTypeNumber(b))
якщо так, всередині умови перевіряємо чи дільник "b" строго не дорівнює (!==) 0, адже ділення на 0 виведе значення "Infinite":
if (b !== 0)
і аж лише тоді якщо вона виконується, виводимо результат ділення:
return divisionText += a / b

але, якщо хоча б одна з цих умов не виконується (тип даних вказаний рядковий, чи дільник = 0), то виводимо (return) в консоль текст "Результат ділення: ділення неможливе" (divisionText += divisionTextError)

function showMessage(text = "Нічого не вказано") { console.log(text)
}

let checkTypeNumber = (someVar) => typeof someVar == "number" ? someVar : false

function calcDivision(a = 0, b = 0) {
let divisionText = "Результат ділення: "
let divisionTextError = "ділення неможливе"
if (checkTypeNumber(a) && checkTypeNumber(b)) {
if (b !== 0) {
return divisionText += a / b
}
else {
return divisionText += divisionTextError }
}
else {
return divisionText += divisionTextError }
}

showMessage(calcDivision(6, 3))
showMessage(calcDivision(6, "3"))
showMessage(calcDivision(6, false))
showMessage(calcDivision(6, 0))
showMessage(calcDivision(6))

Задача №5

Створіть масив даних - 5 елементів, один з яких число 10 Обробіть масив за допомогою методу перебору Перевіряйте елемент на відповідність числу 10, та у разі відповідності, виводьте в консоль

Розв'язок:

Задаємо масив даних "score" зі п'ятьма значеннями:
"10" (текст "10"),
"десять" (текст "десять"),
10 (число 10),
"5" + 5 (текст "5" до якого приєднуємо число 5 = 55 (тут оператор "+" є унарним, тобто це не математичне додавання)),
5 * 3 (число 15).

let score = ["10", "десять", 10, "5" + 5, 5 * 3]

І варіант. Використовуємо метод перебору масиву "forEach".
Тобто, якщо (if) значення (value) === (строго дорівнює) 10,
то виводимо в консоль це значення (console.log(value))
і номер комірки (console.log(index))

let score = ["10", "десять", 10, "5" + 5, 5 * 3]

score.forEach((value, index) => {
if (value === 10) {
console.log(index)
console.log(value)
}
})

ІІ варіант. Використовуємо includes.
Тобто, якщо (if) масив (score) включає (includes) число 10,
то виводимо в консоль текст "Вітаю! 10 з 10",
якщо не включає - текст "Все пропало!"

let score = ["10", "десять", 10, "5" + 5, 5 * 3]

if (score.includes(10)) {
console.log("Вітаю! 10 з 10")
}
else {
console.log("Все пропало!")
}

ІІІ варіант. Використовуємо includes тільки з оператором умови ?, що скорочує код.
Тобто, чи масив (score) включає (includes) число 10 ?,
якщо так, то виводимо в консоль текст "Знову 10 з 10",
якщо не включає - текст "Все пропало!"

let score = ["10", "десять", 10, "5" + 5, 5 * 3]

score.includes(10) ? console.log("Знову 10 з 10") : console.log("Все пропало!")
